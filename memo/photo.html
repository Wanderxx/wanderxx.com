var tablecolor="rgb(255,255,255)";//桌面颜色
var cardrad=30;//多边型半径
var deck=[];//一副牌数组sy
var firstsx=30;//第一张牌位置
var firstsy=50;
var margin=30;//牌间距
var cardwidth=100;
var cardheight=100;
var matched;//
var starttime;//计算消耗时
var count=0;
var px;//鼠标坐标全局变量
var py;
var st=[
["me.png","me1.png",false],
["litao.png","litao1.png",false],
["linpin.png","linpin1.png",false],
["jiaqi.png","jiaqi1.png",false],

function Card(sx,sy,swidth,sheight,img,info){
this.img=img;
this.sx=sx;
this.sy=sy;
this.swidth=swidth;
this.sheight=sheight;
this.info=info;//多边形边数
this.draw=drawback;
}

function makedeck(){
var i;
var n;
var c;
var acard;//保存一对牌中第一张变量
var bcard;
var pica;//图片对象
var picb;
var cx=firstsx;//保存第一个位置
var cy=firstsy;
for(n=0;n<5;n++){
do{c=Math.floor(Math.random()*st.length);}//寻找为选择的对
while(st[c][2]==true)
st[c][2]=true;
pairs.push(st[c]);
// ctx.drawImage(st[c][0],cx,cy,cardwidth,cardheight);
// ctx.drawImage(st[c][1],cx,cy+cardheight+margin,cardwidth,cardheight);
// cx=cx+cardwidth+margin;
}

for(i=0;i<pairs.length;i++){
pica =new Image();
pica.src=pairs[i][0];
picb =new Image();
picb.src=pairs[i][1];
acard=new Card(cx,cy,cardwidth,cardheight,pica,i);//创建一张牌
bcard=new Card(cx,cy+cardheight+margin,cardwidth,cardheight,picb,i);//同样信息，第二张，位于第一张下方
deck.push(acard);
deck.push(bcard);
cx=cx+cardwidth+margin;//下一个多边型，由此循环

function shuffle(){
var i;
var k;
var holderinfo;
var holderimg;
var dl=deck.length;//保存牌数目变量
var n;//交换数索引
for (n=0;n<3*dl;n++){//洗牌次数，为牌数*3
i=Math.floor(Math.random()*dl);//得到随机牌
k=Math.floor(Math.random()*dl);//再得到一随机牌
holderinfo=deck[i].info; //经典交换算法
holderimg=deck[i].img;
deck[i].info=deck[k].info;
deck[i].img=deck[k].img;
deck[k].info=holderinfo;
deck[k].img=holderimg;
}
}

function drawback(){
ctx.fillStyle=backcolor;
ctx.fillRect(this.sx,this.sy,this.swidth,this.sheight);
}
function mouseMove(ev){ //得到鼠标左边函数
ev = ev || window.event;
var mousePos = mouseCoords(ev);

px= mousePos.x;
py= mousePos.y;
}

function mouseCoords(ev)
{
if(ev.pageX || ev.pageY){
return {x:ev.pageX, y:ev.pageY};
}
return{
x:ev.clientX + document.body.scrollLeft - document.body.clientLeft,
y:ev.clientY + document.body.scrollTop - document.body.clientTop
};
}
document.onmousemove = mouseMove;
function choose(ev){
var mx=px;
var my=py;
var i;
for(i=0;i<deck.length;i++){//循环处理整幅牌
var card=deck[i];//抽出一个牌意引用简化代码
if(card.sx>=0){ //检查此牌是否标志以删除
if
((mx>card.sx)&&(mx<card.sx+card.swidth)&&(my>card.sy)&&(my<card.sy+card.sheight)){//检查鼠标是否在牌上
if((firstpick) || (i!=firstcard)) break; //检查玩家是否再次单击第一张牌，不是或者，第一次点击，跳出循环
}
}
}
if(i<deck.length){
if(firstpick){//若di一次点击
firstcard=i;
firstpick=false;
ctx.drawImage(card.img,card.sx,card.sy,card.swidth,card.sheight);
}
else{
secondcard=i;//第2张牌
ctx.drawImage(card.img,card.sx,card.sy,card.swidth,card.sheight);
if(deck[secondcard].info==deck[firstcard].info){//检察第二张和第一张是否匹配
matched=true;
count++;
ctx.fillStyle=tablecolor;
ctx.fillRect(10,340,900,100);//清空要写文本区域
ctx.fillStyle=backcolor;
ctx.fillText("You Have matched:"+String(count)+"pairs",10,360);
if(count>=.5*deck.length){//是否全部匹配
var now =new Date();
now=Number(now.getTime());
var seconds=Math.floor(.5+(now-starttime)/1000);//计算经过多少秒
ctx.fillStyle=tablecolor;
ctx.clearRect(0,0,900,400);
ctx.fillStyle=backcolor;
ctx.fillText("You finished in "+String(seconds)+" seconds.",10,100);
ctx.fillText("嘿嘿，你真棒！刷新再玩一次？",10,300);
}
}
else {//不匹配
matched=false;
}
firstpick=true;
setTimeout(flipback,1000);//暂停
}
}
}

function flipback(){
if(!matched){//不匹配画背面
deck[firstcard].draw();
deck[secondcard].draw();
}
else{//匹配则覆盖
ctx.fillStyle=tablecolor;
ctx.fillRect(deck[secondcard].sx,deck[secondcard].sy
,deck[secondcard].swidth,deck[secondcard].sheight);
ctx.fillRect(deck[firstcard].sx,deck[firstcard].sy
,deck[firstcard].swidth,deck[firstcard].sheight);
deck[secondcard].sx=-1;//设置被移除
deck[firstcard].sx=-1;
}
}

makedeck();//创建牌
shuffle();
ctx.font="bold 20pt sans-serif";
ctx.fillText("随机照片已生成，选中两张同一人照片匹配",10,30);
ctx.fillText("请一次点击2张照片 You Have matched: "+0+" pairs",10,360);
starttime=new Date();
starttime=Number(starttime.getTime());
}
